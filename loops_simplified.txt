// for Loop (Counting Loop)

// This loop runs a specific number of times.

#include <stdio.h>

int main() {
    // Print numbers from 1 to 5
    for (int i = 1; i <= 5; i++) {
        printf("Number: %d\n", i);
    }
    return 0;
}
/*
Output:
Number: 1  
Number: 2  
Number: 3  
Number: 4  
Number: 5  
*/
//----------------------------------------------------
// while Loop (Condition-Based Loop)
// This loop runs as long as the condition is true.

#include <stdio.h>

int main() {
    int i = 1;

    // Print numbers from 1 to 5
    while (i <= 5) {
        printf("Number: %d\n", i);
        i++;  // Increment i
    }
    return 0;
}
//----------------------------------------------------
// do-while Loop (Runs at Least Once)
// This loop always runs at least once, even if the condition is false.
#include <stdio.h>

int main() {
    int i = 1;

    do {
        printf("Number: %d\n", i);
        i++;
    } while (i <= 5);

    return 0;
}

// Nested for Loop (Multiplication Table)
// A for loop inside another for loop to print a multiplication table.
#include <stdio.h>

int main() {
    for (int i = 1; i <= 3; i++) { // Outer loop (rows)
        for (int j = 1; j <= 3; j++) { // Inner loop (columns)
            printf("%d x %d = %d\t", i, j, i * j);
        }
        printf("\n"); // New line after each row
    }
    return 0;
}
OUTPUT:
1 x 1 = 1   1 x 2 = 2   1 x 3 = 3  
2 x 1 = 2   2 x 2 = 4   2 x 3 = 6  
3 x 1 = 3   3 x 2 = 6   3 x 3 = 9  
//----------------------------------------------------
// Nested while Loop (Triangle Pattern)
// A while loop inside another while loop to print a triangle pattern.
#include <stdio.h>

int main() {
    int i = 1;

    while (i <= 3) { // Outer loop (rows)
        int j = 1;
        while (j <= i) { // Inner loop (columns)
            printf("* ");
            j++;
        }
        printf("\n"); // Move to the next line
        i++;
    }
    return 0;
}
OUTPUT:
*  
* *  
* * *

// Nested do-while Loop (Number Grid)
// A do-while loop inside another do-while loop to create a number grid.
#include <stdio.h>

int main() {
    int i = 1;

    do {
        int j = 1;
        do {
            printf("%d ", j);
            j++;
        } while (j <= 3); // Inner loop (columns)

        printf("\n"); // Move to the next row
        i++;
    } while (i <= 3); // Outer loop (rows)

    return 0;
}
OUTPUT:
1 2 3  
1 2 3  
1 2 3

// Three Nested for Loops (3D Matrix Output)
// This prints a 3D matrix (like a cube with layers).
#include <stdio.h>

int main() {
    for (int i = 1; i <= 2; i++) { // Outer loop (Layers)
        for (int j = 1; j <= 3; j++) { // Middle loop (Rows)
            for (int k = 1; k <= 3; k++) { // Inner loop (Columns)
                printf("Layer %d, Row %d, Col %d\n", i, j, k);
            }
        }
        printf("\n"); // Separate layers
    }
    return 0;
}

// Three Nested while Loops (Star Pyramid)
// This prints a pyramid of stars using three while loops.
#include <stdio.h>

int main() {
    int i = 1;

    while (i <= 3) { // Outer loop (Levels)
        int j = 1;
        while (j <= i) { // Middle loop (Rows)
            int k = 1;
            while (k <= j) { // Inner loop (Stars)
                printf("* ");
                k++;
            }
            printf("\n");
            j++;
        }
        printf("\n"); // Separate levels
        i++;
    }
    return 0;
}
Output:
*  

*  
* *  

*  
* *  
* * *

// Three Nested do-while Loops (Number Pyramid)
// This prints a number pyramid using do-while loops.
#include <stdio.h>

int main() {
    int i = 1;

    do {
        int j = 1;
        do {
            int k = 1;
            do {
                printf("%d ", j);
                k++;
            } while (k <= j); // Inner loop (Numbers)
            printf("\n");
            j++;
        } while (j <= i); // Middle loop (Rows)
        printf("\n"); // Separate levels
        i++;
    } while (i <= 3); // Outer loop (Levels)

    return 0;
}
